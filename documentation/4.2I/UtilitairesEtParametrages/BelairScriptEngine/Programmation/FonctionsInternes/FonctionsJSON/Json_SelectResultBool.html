Fonctions JSON

Json:SelectResultBool(doc,path,default)

cette fonction sélectionne la valeur comme Json:SelectResult puis la convertit en type booléen comme Json:GetCurrentValueBool.


Paramètres d'entrée :

  • doc : un objet de type 'document JSON'
  • path : la requête JSONPath
  • default : la valeur par défaut si la requête ne trouve aucun résultat (optionnel)

Résultat :
la valeur convertie en booléen.


Exemple : 

local doc
doc := Json:New()
Json:NewObject(doc, "Cars")
  Json:NewArray(doc, "Renault")
    Json:NewObject(doc)
      Json:NewValue(doc, "Model", "clio")
      Json:NewValue(doc, "Price", 10990)
      Json:NewValue(doc, "Currency", "EUR")
      Json:NewValue(doc, "InStock", false)
    Json:EndObject(doc)
    Json:NewObject(doc)
      Json:NewValue(doc, "Model", "zoe")
      Json:NewValue(doc, "Price", 25700)
      Json:NewValue(doc, "Currency", "EUR")
      Json:NewValue(doc, "InStock", true)
    Json:EndObject(doc)
  Json:EndArray(doc)
Json:EndObject(doc)
Json:Close(doc)            
print(Json:SelectResultBool(doc, "cars.renault[0].InStock"))
print(Json:SelectResultBool(doc, "cars.renault[1].InStock"))
print(Json:SelectResultBool(doc, "NotAGoodPath", false))   
return 0

Résultat : 

False
True
False
