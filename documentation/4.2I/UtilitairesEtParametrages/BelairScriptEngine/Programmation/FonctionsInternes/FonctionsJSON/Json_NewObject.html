Fonctions JSON

Json:NewObject(doc, name)

La fonction ajoute un enfant de type JSONObject.


Paramètres d'entrée :

  • doc : un objet de type 'document JSON'
  • name : le nom du JSONObject (optionnel)

Résultat :
Aucun.


Exemple : 

local doc
doc := Json:New()
Json:NewObject(doc, "colors")
  Json:NewArray(doc, "color")
    Json:NewObject(doc)
      Json:NewValue(doc, "color1", "Red")
    Json:EndObject(doc)
    Json:NewObject(doc)
      Json:NewValue(doc, "color2", "Green")
    Json:EndObject(doc)
    Json:NewObject(doc)
      Json:NewValue(doc, "color3", "Blue")
    Json:EndObject(doc)
  Json:EndArray(doc)
Json:EndObject(doc)
Json:Close(doc)            
return Json:PrettyFormat(doc)

Résultat : 

{
  "colors": {
    "color": [
      {
        "color1": "Red" 
      },
      {
        "color2": "Green" 
      },
      {
        "color3": "Blue" 
      }
    ]
  }
}
