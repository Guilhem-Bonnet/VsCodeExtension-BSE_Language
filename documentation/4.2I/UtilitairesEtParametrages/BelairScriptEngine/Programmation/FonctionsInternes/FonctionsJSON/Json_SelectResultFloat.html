Fonctions JSON

Json:SelectResultFloat(doc,path,default)

cette fonction sélectionne la valeur comme Json:SelectResult puis la convertit en type numérique virgule flottante comme Json:GetCurrentValueFloat.


Paramètres d'entrée :

  • doc : un objet de type 'document JSON'
  • path : la requête JSONPath
  • default : la valeur par défaut si la requête ne trouve aucun résultat (optionnel)

Résultat :
la valeur convertie en type numérique.


Exemple : 

local doc
doc := Json:New()
Json:NewObject(doc, "Cars")
  Json:NewArray(doc, "Renault")
    Json:NewObject(doc)
      Json:NewValue(doc, "Model", "clio")
      Json:NewValue(doc, "Price", 10990.987)
      Json:NewValue(doc, "Currency", "EUR")
      Json:NewValue(doc, "InStock", false)
      Json:NewValue(doc, "Production_Date", EncodeDateTime(2012, 01, 01, 10, 00, 00))
    Json:EndObject(doc)
    Json:NewObject(doc)
      Json:NewValue(doc, "Model", "zoe")
      Json:NewValue(doc, "Price", 25700.999)
      Json:NewValue(doc, "Currency", "EUR")
      Json:NewValue(doc, "InStock", true)
      Json:NewValue(doc, "Production_Date", EncodeDate(2013, 01, 01))
    Json:EndObject(doc)
  Json:EndArray(doc)
Json:EndObject(doc)
Json:Close(doc)            
print(Json:SelectResultFloat(doc, "cars.renault[0].Price"))
print(Json:SelectResultFloat(doc, "cars.renault[1].Price"))
print(Json:SelectResultFloat(doc, "NotAGoodPath", 12.664))   
return 0

Résultat : 

10990,987
25700,999
12,664
