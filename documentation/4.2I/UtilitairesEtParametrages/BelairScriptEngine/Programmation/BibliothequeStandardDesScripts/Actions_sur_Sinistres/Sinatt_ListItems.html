Bibliothèque standard des scripts

Sinatt_ListItems

Listes des situations d'attente d'un sinistre

Parametres

  • sinistre (int, obligatoire) : numéro de sinistre
  • all (boolean, optionnel) : si TRUE, toutes les situations ; si FALSE (par défaut), seules les non traitées
  • statut (string, optionnel) : filtre sur le code statut

Retour

Flux XML

<?xml version="1.0"?>
<sinatt sinistre="24">
  <objects>
    <object typename="sinatt">
      <param name="Codeatt" type="ptString">03</param>
      <param name="Commen" type="ptString"/>
      <param name="Dateatt" type="ptDateTime" date_val="2014-09-10T00:00:00.000"/>
      <param name="external_tab_statut" type="ptString">Reçu incomplet</param>
      <param name="Ficage" type="ptInt" int_val="-1"/>
      <param name="Modifpar" type="ptString">SUPERVISEUR</param>
      <param name="Recue" type="ptUnknown" is_null="true"/>
      <param name="Sinistre" type="ptInt" int_val="24"/>
      <param name="Statut" type="ptString">3</param>
      <param name="Traite" type="ptUnknown" is_null="true"/>
    </object>
  </objects>
</sinatt>
