Societe_ListKcoItems

Liste des comptabilités existantes (associées ou non à une entité comptable) OU résumé de l'utilisation des codes

Parametres

  • Summary (Bool, optionnel)

Résultat

Si summary = True

<?xml version="1.0"?>
<kco-codes>
  <used>12A</used>
  <free>3456789BCDEFGHIJKLMNOPQRSTUVWXYZ</free>
  <firstavailable>3</firstavailable>
</kco-codes>

Si summary = False

<?xml version="1.0"?>
<kco-rows>
  <kco>
    <codek>1</codek>
    <entite>1</entite>
    <libk>Comptabilité Modèle n. 1 Bel Air</libk>
    <siren>357465454546541</siren>
    <orias>94039849</orias>
  </kco>
  <kco>
    <codek>2</codek>
    <entite>2</entite>
    <libk>Comptabilité NUMERO 2 Bel Air</libk>
    <siren isnull="true"/>
    <orias isnull="true"/>
  </kco>
  <kco>
    <codek>A</codek>
    <entite>0</entite>
    <libk>Compta A</libk>
    <siren>357465454546541</siren>
    <orias isnull="true"/>
  </kco>
</kco-rows>
